	function saveHistoric() {

		var url = window.location.href;
		var title = document.title;
		if (document.querySelectorAll('meta[property="og:image"]')[0] !== undefined) {
			var image = document.querySelectorAll('meta[property="og:image"]')[0].content;
		} else {
			var image = BASE + '/img/logo-site.png';
		}

		var day = new Date();
		day.toLocaleDateString();

		var urls = JSON.parse(localStorage.getItem('links') || '[]');
		var count = parseInt(Object.keys(urls).length);

		if (count > 299) {
			console.log('alfo');
			urls.splice(0, 1);
		}

		for (var i = 0; i < urls.length; i++) {
			if (urls[i].url === url) {
				urls.splice(i, 1);
			}
		}

		urls.push({
			url: url,
			title: title,
			image: image,
			day: day
		});

		localStorage.setItem("links", JSON.stringify(urls));
	}
